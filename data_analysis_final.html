import scipy.stats as stats
import numpy as np 
import matplotlib.pyplot as plt 
import seaborn as sns 


def SubjectStats(fileName):
    
    # read from csv file
    df = pandas.read_csv(fileName)
    
    # obtain the subject number in two digits 
    subject = f"{df.loc[0,'subject']:02d}"
    
    # 4 conditions 
    # separation: 0: not separate, or move together; 1: move separately 
    # offset: 0: simultaneous offset; 1: delayed offset 
    sep_idx = [0,0,1,1] 
    offset_idx = [0,1,0,1]  
    labels = ['notSep_simultaneous', 'notSep_delayed', 'sep_simultaneous', 'sep_delayed']
    
    # return values
    error_mean = []
    rt_mean = []
    
    # set up parameters for 2 plots
    fig, axs = plt.subplots(ncols = 2, figsize = (11.7, 8.27))
    axs[0].set_ylim(0,100)   # error rate ranges from 0 to 100
    fig.suptitle(fileName)
    
    # create a loop to run through all conditions
    # the range is 4 because there are 4 conditions 
    for i in range(4):
        # calcuate the mean error rate
        df1 = df.loc[(df.loc[:, 'sep'] == sep_idx[i]) & (df.loc[:, 'offset'] == offset_idx[i]), :]
        error_mean.append(np.mean(df1.loc[:, 'error']))
        sns.distplot(df1.loc[:, 'error'], label = labels[i], hist = False, ax = axs[0])

        # calculate the mean response time (rt) with the correct trials only 
        df1 = df1.loc[df1.loc[:, 'error'] == 0, :]
        rt_mean.append(np.mean(df1.loc[:, 'rt']))
        sns.distplot(df1.loc[:, 'rt'], label = labels[i], hist = False, ax = axs[1])
    
    
    plt.legend()
    plt.show()
        
    return subject, error_mean, rt_mean

# to follow up on the significant interaction of sep and offset (p < 0.05)
# conduct a one-sample t-test on the masking effect. Masking effect = delayed offset - simultaneous offset

notSep_delayed = df.loc[(df.loc[:, 'sep'] == 'notSep') & (df.loc[:, 'offset'] == 'delayed') , 'arcsin_error_mean'] 
notSep_simultaneous = df.loc[ (df.loc[:, 'sep'] == 'notSep') & (df.loc[:, 'offset'] == 'simultaneous'), 'arcsin_error_mean']
sep_delayed = df.loc[(df.loc[:, 'sep'] == 'sep') & (df.loc[:, 'offset'] == 'delayed') , 'arcsin_error_mean'] 
sep_simultaneous = df.loc[ (df.loc[:, 'sep'] == 'sep') & (df.loc[:, 'offset'] == 'simultaneous'), 'arcsin_error_mean']

# note: # cannot just do: notSep_effect = notSep_delayed - notSep_simultaneous because they have different indices for the row
notSep_effect = notSep_delayed.to_numpy() - notSep_simultaneous.to_numpy()
sep_effect = sep_delayed.to_numpy() - sep_simultaneous.to_numpy()

# conduct a one-sample t-test: 
t1, p1 = stats.ttest_1samp(notSep_effect,0)
print('notSep masking effect - t value is: ' + f"{t1:.4f}" + ', '+ 'p value is: ' + f"{p1:.4f}")
t2, p2 = stats.ttest_1samp(sep_effect,0)
print('sep masking effect - t value is: ' + f"{t2:.4f}" + ', '+ 'p value is: ' + f"{p2:.4f}")
